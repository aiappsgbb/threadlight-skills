## Edit type
`insert`

## Edit recipe
1. Create or open `infra/observability/alerts.bicep`.
2. If the file is empty or does not exist, insert baseline alert rules:

```bicep
param appInsightsResourceId string
param appInsightsName string
param actionGroupResourceId string
param location string
param tags object = {}

// Alert: High error rate (5xx errors in last 5 minutes)
resource highErrorRateAlert 'Microsoft.Insights/metricAlerts@2018-03-01' = {
  name: '${appInsightsName}-high-error-rate'
  location: 'global'
  properties: {
    description: 'Agent invocation error rate exceeds 5% in last 5 minutes'
    severity: 1
    enabled: true
    scopes: [appInsightsResourceId]
    evaluationFrequency: 'PT5M'
    windowSize: 'PT5M'
    criteria: {
      'odata.type': 'Microsoft.Azure.Monitor.MultipleResourceMultipleMetricCriteria'
      allOf: [
        {
          name: 'Percentage'
          metricName: 'failedRequests'
          operator: 'GreaterThan'
          threshold: 5
          timeAggregation: 'Total'
        }
      ]
    }
    actions: [
      {
        actionGroupId: actionGroupResourceId
      }
    ]
  }
  tags: tags
}

// Alert: Slow requests (p95 latency > 5s in last 5 minutes)
resource slowRequestAlert 'Microsoft.Insights/metricAlerts@2018-03-01' = {
  name: '${appInsightsName}-slow-requests'
  location: 'global'
  properties: {
    description: 'Request latency p95 exceeds 5 seconds'
    severity: 2
    enabled: true
    scopes: [appInsightsResourceId]
    evaluationFrequency: 'PT5M'
    windowSize: 'PT5M'
    criteria: {
      'odata.type': 'Microsoft.Azure.Monitor.MultipleResourceMultipleMetricCriteria'
      allOf: [
        {
          name: 'Latency'
          metricName: 'performancecounters/processorCpuPercentage'
          operator: 'GreaterThan'
          threshold: 5000
          timeAggregation: 'Average'
        }
      ]
    }
    actions: [
      {
        actionGroupId: actionGroupResourceId
      }
    ]
  }
  tags: tags
}

output alertRuleIds array = [
  highErrorRateAlert.id
  slowRequestAlert.id
]
```

3. Reference this module from the main template and pass `appInsightsResourceId` and `actionGroupResourceId`.
4. Adjust thresholds based on production SLA (e.g., 5% error rate, 5s latency) — these are conservative starting points.
